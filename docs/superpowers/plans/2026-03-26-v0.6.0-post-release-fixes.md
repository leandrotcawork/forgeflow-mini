- [ ] **Step 3: Add `isSafeKeyPath` helper and guard in `setDeep` and `brainConfigWrite`**

In `mcp/brain-config-server.js`, add this function immediately after `splitKeyPath` (~line 228):

```js
var DANGEROUS_KEY_SEGMENTS = ['__proto__', 'constructor', 'prototype'];

/**
 * Returns false if any segment of the key path is a prototype-poisoning candidate.
 */
function isSafeKeyPath(keyPath) {
  var parts = splitKeyPath(keyPath);
  for (var i = 0; i < parts.length; i++) {
    if (DANGEROUS_KEY_SEGMENTS.indexOf(parts[i]) !== -1) return false;
  }
  return true;
}
```

Then update `setDeep` to guard at entry (lines 249-259):

```js
function setDeep(obj, keyPath, value) {
  if (!isSafeKeyPath(keyPath)) {
    return; // silently refuse — callers should validate before calling setDeep
  }
  var parts = splitKeyPath(keyPath);
  var current = obj;
  for (var i = 0; i < parts.length - 1; i++) {
    if (current[parts[i]] === undefined || current[parts[i]] === null || typeof current[parts[i]] !== 'object') {
      current[parts[i]] = {};
    }
    current = current[parts[i]];
  }
  current[parts[parts.length - 1]] = value;
}
```

Then add a guard at the top of `brainConfigWrite`, immediately after the `var key = args.key;` line (~line 444):

```js
  if (!isSafeKeyPath(key)) {
    return error('Invalid key "' + key + '": key segments cannot be __proto__, constructor, or prototype.');
  }
```
